#
#	Set up third-party dependencies for libwfa
#

include(CheckCXXSourceCompiles)

#   Test drand48
#
check_cxx_source_compiles("
#include<cstdlib>
int main() {
  double d = drand48();
  return 0;
}
" HAVE_DRAND48)
if(HAVE_DRAND48)
    add_definitions(-DHAVE_DRAND48)
endif(HAVE_DRAND48)


#
#	Find armadillo
#
set(CMAKE_MODULE_PATH_SAVED ${CMAKE_MODULE_PATH})
set(CMAKE_MODULE_PATH ${LIBWFA_DIR}/cmake)

set(ARMADILLO_PATH "" CACHE PATH 
	"Path of Armadillo C++ linear algebra library (only set if non-standard)")	
find_package(Armadillo)
set(CMAKE_MODULE_PATH ${CMAKE_MODULE_PATH_SAVED})
if(NOT ARMADILLO_FOUND) 
	message(FATAL_ERROR "Armadillo library not found")
endif(NOT ARMADILLO_FOUND) 

# TODO: check which min version is required
if("${ARMA_VERSION}" VERSION_GREATER "3.6.2") 
	message(FATAL_ERROR 
		"Armadillo library is too old. Please upgrade to >= 3.6.2")
endif()	 

#
#	libwfa dependencies
#
message(STATUS "ARMADILLO_INCLUDE_DIRS: ${ARMADILLO_INCLUDE_DIRS}")
message(STATUS "ARMADILLO_LIBRARIES: ${ARMADILLO_LIBRARIES}")

include_directories(${ARMADILLO_INCLUDE_DIRS})
set(LIBWFA_DEP_LIBS ${ARMADILLO_LIBRARIES})
